def knapsack(W,wt,val,n):
    if n==0 or W==0:
        return 0
    if (wt[n-1]>W):
        return knapsack(W,wt,val,n-1)
    else:
        return max(val[n-1]+knapsack(W-wt[n-1],wt,val,n-1),knapsack(W,wt,val,n-1))
val=[30,40,50]
wt=[10,20,30]
W=50
n=len(val)
print('Values:',val)
print('Weights:',wt)
print('Knapsack Capacity:',W)
print('Maximum weight:',knapsack(W,wt,val,n))
